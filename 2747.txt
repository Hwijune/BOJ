import java.io.*;
import java.util.*;

public class Main
{
	static int fibonacci(int n)
	{
		// F(0) = 0 and F(1) = 1
		// 재귀 함수가 끝나는 종료조건
		if (n == 0 || n == 1)
			return n;
		// F(n) = F(n-1) + F(n-2)
		// 재귀 함수에서 재귀되는 부분
		else
			return fibonacci(n - 1) + fibonacci(n - 2);
	}
	
	public static void main(String... args) throws Exception
	{
		Scanner in = new Scanner(System.in);
		int n = in.nextInt();
		System.out.println(fibonacci(n));
	}
}
