#include<cstdio>
#include<vector>
#include<queue>
using namespace std;
 
vector<int> deg;
vector<int> edge[32002];
 
int main() {
    int n, m;
    scanf("%d%d", &n, &m);
    deg = vector<int>(n + 1, 0);
    for (int i = 0; i < m; i++) {
        int u, v;
        scanf("%d%d", &u, &v);
        edge[u].push_back(v);
        deg[v]++; //Â÷¼ö
    }
 
    queue<int> Q;
    for (int i = 1; i <= n; i++)
        if (deg[i] == 0) 
            Q.push(i);
 
    while (!Q.empty()) {
        int here = Q.front(); 
        Q.pop();
        printf("%d ", here);
 
        for (auto iter = edge[here].begin(); iter != edge[here].end(); iter++) 
        {
            int there = *iter;
            deg[there]--;
            if (deg[there] == 0)    
                Q.push(there);
        }
    }
    return 0;
}
